def process_api_response(prompt, max_length: 1000, retries: 3)
  attempts = 0
  begin
    response = fetch_api_response(prompt, max_length: max_length)
    
    if response[:finish_reason] == "length"
      if attempts < retries
        attempts += 1
        max_length = max_length - 100
        return process_api_response(prompt, max_length: max_length, retries: retries)
      else
        raise "Max retries reached with incomplete response"
      end
    end
    
    parse_response(response[:content])
  rescue => e
    puts "Error: #{e.message}"
    nil
  end
end
