```ruby
return  unless stmts && !stmts.empty?
last = stmts.last
case last
when MIR::ReturnStmt
  true
when MIR::RawZig
  last.code.to_s.include?("return;") || last.code.to_s.include?("return ")
when MIR::ScopeBlock
  flow_body_terminates?(last.body || )
when MIR::IfStmt
  flow_body_terminates?(last.then_body || []) &&
    (last.else_body && !last.else_body.empty? && flow_body_terminates?(last.else_body || []))
else
  false
end
```
