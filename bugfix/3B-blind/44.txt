```ruby
def borrow(ownership_graph, node_id)
  # Find the node with the given node_id
  node = ownership_graph.find { |n| n.id == node_id }

  # If the node is not found, return nil
  return nil if node.nil?

  # Get the list of nodes that borrow from this node
  borrowers = ownership_graph.select { |n| n.borrowers.include?(node) }

  # Return the list of borrowers
  borrowers
end
```
