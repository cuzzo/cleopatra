```ruby
def initialize(level:, message:, token:, category:, fixes:)
    raise ArgumentError, "bad level #{level.inspect}" unless FixableFinding::LEVELS.include?(level)
    raise ArgumentError, "bad category #{category.inspect}" unless FixableFinding::CATEGORIES.include?(category)
    @level = level
    @message = message
    @token = token
    @category = category
    @fixes = Array(fixes)
    raise ArgumentError, "at least one Fix required" if @fixes.empty?
  end
```
